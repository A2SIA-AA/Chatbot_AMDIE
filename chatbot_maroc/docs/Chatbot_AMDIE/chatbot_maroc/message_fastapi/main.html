 63@app.middleware("http")
 64async def log_requests(request, call_next):
 65    """Log automatique de toutes les requêtes"""
 66    #mesure du temp d'exécution
 67    start_time = time.time()
 68    response = await call_next(request)
 69    process_time = time.time() - start_time
 70
 71    #enregistrement des infos utiles
 72    logger.info(
 73        f"{request.method} {request.url} - "
 74        f"Status: {response.status_code} - "
 75        f"Time: {process_time:.4f}s"
 76    )
 77
 78    #ajout des infos dans l'en-tête de la requête
 79    response.headers["X-Process-Time"] = str(process_time)
 80    return response

251@app.post("/api/v1/auth/login", response_model=LoginResponse, tags=["Authentication"])
252async def login(login_data: LoginRequest):
253    """Connexion utilisateur avec JWT"""
254    user = authenticate_user(login_data.email, login_data.password)
255    if not user:
256        raise HTTPException(
257            status_code=401,
258            detail="Email ou mot de passe incorrect"
259        )
260
261    access_token_expires = timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
262    access_token = create_access_token(
263        data={"sub": user["email"]}, expires_delta=access_token_expires
264    )
265
266    user_response = User(
267        username=user["username"],
268        email=user["email"],
269        role=user["role"],
270        permissions=user["permissions"],
271        full_name=user["full_name"],
272        department=user["department"]
273    )
274
275    return LoginResponse(
276        access_token=access_token,
277        token_type="bearer",
278        user=user_response,
279        expires_in=ACCESS_TOKEN_EXPIRE_MINUTES * 60  # en secondes
280    )

321
322        question = request.get("question", "")
323        if not question.strip():
324            raise HTTPException(status_code=400, detail="Question requise")
325
326        # Générer session ID avec info utilisateur
327        session_id = f"session_{current_user['role']}_{int(time.time())}_{os.urandom(4).hex()}"

329        # Message de démarrage avec info utilisateur
330        await message_store.add_message(session_id, {

339

342        script_path = os.path.join(backend_path, "chatbot_wrapper.py")
343
344        if not os.path.exists(script_path):
345            raise HTTPException(status_code=500, detail=f"Backend non trouvé: {script_path}")
346
347        # LANCER LE BACKEND avec les permissions utilisateur
348        try:
349            # permissions_csv
350            permissions_csv = ",".join(current_user['permissions']) if isinstance(current_user['permissions'], list) else str(current_user['permissions'])

356                    session_id=session_id,

369                'type': 'progress',
370                'content': f'Backend IA initialisé avec permissions {current_user["role"]}',
371                'metadata': {
372                    'source': 'api_orchestrator',
373                    'user_role': current_user['role']
374                }
375            })

386        return {
387            "sessionId": session_id,
388            "status": "processing_started",
389            "message": "Traitement IA démarré avec succès",

391                "role": current_user['role'],

490# Fonction commune pour traiter le callback
491async def process_keycloak_callback(code: str, redirect_uri: str):
492    """Traitement commun du callback Keycloak"""
493    try:
494        # Échanger le code contre un token
495        token_response = exchange_code_for_token(code, redirect_uri)
496
497        # Décoder le token pour avoir les infos utilisateur
498        from jose import jwt
499        from auth_keycloack import keycloak_openid
500
501        public_key = keycloak_openid.public_key()
502        keycloak_public_key = f"-----BEGIN PUBLIC KEY-----\n{public_key}\n-----END PUBLIC KEY-----"
503
504        token_info = jwt.decode(
505            token_response["access_token"],
506            keycloak_public_key,
507            algorithms=["RS256"],
508            audience="account"
509        )
510
511        # DEBUG - Afficher les rôles détectés
512        keycloak_roles = token_info.get('realm_access', {}).get('roles', [])
513        logger.info(f"DEBUG - Rôles Keycloak détectés: {keycloak_roles}")
514
515        # Convertir vers le format de ton User model existant
516        from auth_keycloack import keycloak_role_to_amdie_role, get_permissions_from_role
517
518        amdie_role = keycloak_role_to_amdie_role(keycloak_roles)
519        logger.info(f" DEBUG - Rôle AMDIE mappé: {amdie_role}")
520
521        user_response = User(
522            username=token_info.get('preferred_username', 'unknown'),
523            email=token_info.get('email', ''),
524            role=amdie_role,
525            permissions=get_permissions_from_role(amdie_role),
526            full_name=f"{token_info.get('given_name', '')} {token_info.get('family_name', '')}".strip(),
527            department="External" if amdie_role == "public" else "Internal" if amdie_role == "employee" else "Management"
528        )
529
530        logger.info(f" Connexion Keycloak réussie: {user_response.full_name} ({user_response.role})")
531
532        return LoginResponse(
533            access_token=token_response["access_token"],
534            token_type="bearer",
535            user=user_response,
536            expires_in=token_response.get("expires_in", 3600)
537        )
538
539    except Exception as e:
540        logger.error(f"Erreur callback Keycloak: {e}")
541        raise HTTPException(status_code=400, detail=str(e))
