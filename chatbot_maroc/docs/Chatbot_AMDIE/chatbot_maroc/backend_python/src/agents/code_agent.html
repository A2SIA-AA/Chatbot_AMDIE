213    def _generer_guide_sources_csv(self, dataframes: List[pd.DataFrame]) -> str:
214        """
215        Génère un guide décrivant les fichiers sources CSV à partir d'une liste de DataFrames pandas.
216
217        Le guide inclut les informations sur le titre, la source, le contexte et les colonnes clés de
218        chaque DataFrame de la liste.
219
220        :param dataframes: Liste d'objets pandas DataFrame, chacun représentant un ensemble de
221          données avec des attributs facultatifs tels que 'titre', 'source' et 'description'.
222        :type dataframes: List[pd.DataFrame]
223        :return: Une chaîne de texte formatée décrivant les fichiers sources CSV et leurs contextes.
224        :rtype: str
225        """
226
227        guide = ""
228        for i, df in enumerate(dataframes):
229            attrs = getattr(df, 'attrs', {})
230
231            guide += f"""df{i}.csv: {attrs.get('titre', f'Tableau {i}')}
232  Source: {attrs.get('source', 'N/A')}
233  Contexte: {attrs.get('description', 'Données statistiques')}  
234  Colonnes clés: {', '.join(df.columns[:4].astype(str))}
235
236"""
237        return guide

239    def _extraire_sources_utilisees_prompt(self, reponse_brute: str) -> str:
240        """
241        Extrait les sources utilisées à partir d'une réponse brute donnée. La méthode
242        analyse la chaîne de caractères en cherchant des motifs spécifiques qui
243        correspondent à des sources documentées. Si aucune source n'est identifiée,
244        une valeur par défaut est retournée.
245
246        :param reponse_brute: Chaîne de caractères contenant une réponse brute.
247        :return: Sources extraites sous forme de chaîne de caractères.
248        :rtype: str
249        """
250
251        patterns = [
252            r'SOURCES_UTILISEES:\s*\n(.*?)(?=\n[A-Z]|$)',
253            r'SOURCES_UTILISEES:\s*(.*?)(?=\n[A-Z]|$)'
254        ]
255
256        for pattern in patterns:
257            match = re.search(pattern, reponse_brute, re.DOTALL | re.IGNORECASE)
258            if match:
259                return match.group(1).strip()
260
261        return "Voir guide ci-dessus"

263    def _upload_fichier_to_gemini(self, fichiers_csv: List[str], state: ChatbotState):
264        """
265        Télécharge les fichiers CSV spécifiés vers Gemini via l'API fournie par `genai.Client`.
266        Après un téléchargement réussi, les fichiers temporaires sont supprimés.
267
268        :param fichiers_csv: Liste des chemins des fichiers CSV à télécharger.
269        :param state: Contient l'état du chatbot, utilisé pour la journalisation.
270        :return: Une liste des fichiers téléchargés avec succès dans Gemini, ainsi que le
271            client utilisé pour les opérations.
272        :rtype: Tuple[List[Any], genai.Client]
273        :raises Exception: Si une erreur survient lors du téléchargement d'un fichier.
274        """
275        client = genai.Client(api_key=API_KEY)
276
277        fichiers_gemini = []
278        for fichier_csv in fichiers_csv:
279            try:
280                fichier_uploade = client.files.upload(file=fichier_csv)
281                fichiers_gemini.append(fichier_uploade)
282                self.chatbot._log(f"Fichier uploadé: {fichier_csv}", state)
283            except Exception as e:
284                self.chatbot._log_error(f"Erreur upload {fichier_csv}: {str(e)}", state)
285                raise e
286        #suppression immédiate des csvs
287        self._nettoyer_csvs_temporaires(fichiers_csv,state)
288        return fichiers_gemini, client

290    def _creer_csvs_pour_gemini(self, dataframes: List[pd.DataFrame], state: ChatbotState) -> List[str]:
291        """
292        Cette fonction génère des fichiers CSV à partir de plusieurs DataFrames fournis. Pour chaque DataFrame
293        non vide, un fichier CSV est créé avec un nom unique. Un journal des opérations est tenu tout au long du
294        processus. En cas d'erreur, un message d'erreur est enregistré et la fonction retourne une liste vide.
295
296        :param dataframes: Liste de DataFrames à convertir en fichiers CSV. Chaque DataFrame doit être issu de la
297            librairie pandas.
298        :type dataframes: List[pd.DataFrame]
299        :param state: État actuel du Chatbot utilisé pour enregistrer les journaux et les erreurs tout au long du
300            processus d'exécution.
301        :type state: ChatbotState
302        :return: Liste des noms des fichiers CSV créés. Si une erreur survient, une liste vide est retournée.
303        :rtype: List[str]
304        """
305
306        fichiers_csv = []
307
308        try:
309            for i, df in enumerate(dataframes):
310                if df is not None and not df.empty:
311                    nom_fichier = f'df{i}_analysis.csv'
312                    df.to_csv(nom_fichier, index=False, encoding='utf-8')
313                    fichiers_csv.append(nom_fichier)
314                    self.chatbot._log(f"CSV créé: {nom_fichier} {df.shape}", state)
315                else:
316                    self.chatbot._log(f"DataFrame {i} vide ignoré", state)
317
318        except Exception as e:
319            self.chatbot._log_error(f"Erreur création CSV: {str(e)}", state)
320            return []
321
322        return fichiers_csv

325        """
326        Supprime les fichiers CSV temporaires spécifiés et enregistre les logs appropriés
327        dans l'instance de chatbot.
328
329        Cette méthode vérifie d'abord si chaque fichier dans la liste fournie existe dans
330        le système de fichiers local. Si le fichier existe, il est supprimé, et un message
331        de log est enregistré pour confirmer la suppression. En cas d'erreur pendant la
332        suppression d'un fichier, un message d'erreur contenant les détails de l'exception
333        est enregistré dans les logs.
334
335        :param fichiers_csv: Liste des chemins des fichiers CSV temporaires à supprimer.
336        :type fichiers_csv: List[str]
337        :param state: État actuel du chatbot utilisé pour enregistrer les logs.
338        :type state: ChatbotState
339        :return: None
340        """
341
342        for fichier in fichiers_csv:
343            try:
344                if os.path.exists(fichier):
345                    os.remove(fichier)
346                    self.chatbot._log(f"CSV local supprimé: {fichier}", state)
347            except Exception as e:
348                self.chatbot._log_error(f"Erreur suppression {fichier}: {str(e)}", state)

365        for fichier in fichiers_gemini:

367                client.files.delete(name=fichier.name)
368                self.chatbot._log(f"Fichier Gemini supprimé: {fichier.name}", state)
